81 typename std::enable_if<std::is_void<typename std::invoke_result<F>::type>::value, bool>::type invoke(F function, bool sync = false) const {
82 if (!sync) {
83 AbstractTask *_t = new Invocable<void()>::Function(std::forward<F>(function));
84 if (!invokeTask(_t)) {
85 delete _t;
86 return false;
87 }
88 return true;
89 }
90 if (std::this_thread::get_id() == id()) {
91 processTasks();
92 function();
93 return true;
94 }
95 std::atomic_flag finished;
96 AbstractTask *_t = new Invocable<void()>::Function([&function, &finished]() {
97 function();
98 finished.test_and_set();
99 finished.notify_one();
100 });
101 if (!invokeTask(_t)) {
102 delete _t;
103 return false;
104 }
105 for (;;) {
106 finished.wait(false);
107 if (finished.test()) break;
108 }
109 return true;
110 }
