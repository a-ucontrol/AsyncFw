77 template <typename F>
78 typename std::enable_if<std::is_void<typename std::invoke_result<F>::type>::value, bool>::type invoke(F function, bool sync = false) const {
79 if (!sync) {
80 AbstractTask *_t = new Invocable<void()>::Function(std::forward<F>(function));
81 if (!invokeTask(_t)) {
82 delete _t;
83 return false;
84 }
85 return true;
86 }
87 if (std::this_thread::get_id() == id()) {
88 processTasks();
89 function();
90 return true;
91 }
92 std::atomic_flag finished;
93 AbstractTask *_t = new Invocable<void()>::Function([&function, &finished]() {
94 function();
95 finished.test_and_set();
96 finished.notify_one();
97 });
98 if (!invokeTask(_t)) {
99 delete _t;
100 return false;
101 }
102 for (;;) {
103 finished.wait(false);
104 if (finished.test()) break;
105 }
106 return true;
107 }
