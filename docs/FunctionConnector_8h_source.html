63template <typename... Args>
64class FunctionConnector : public AbstractFunctionConnector {
65public:
66 using AbstractFunctionConnector::AbstractFunctionConnector;
68 template <typename T>
69 Connection &operator()(T f, Connection::Type t = Connection::Default) {
70 std::lock_guard<std::mutex> lock(mutex);
71#ifndef __clang_analyzer__
72 return *new Connection(f, this, t);
73#endif
74 }
75
76 template <typename O, typename M>
77 Connection &connect(O *o, M m, Connection::Type t = Connection::Default) {
78 return operator()([o, m](Args... args) { (o->*m)(args...); }, t);
79 }
80
81 void operator()(Args... args) {
82 std::lock_guard<std::mutex> lock(mutex);
83 for (const Connection *c : *reinterpret_cast<std::vector<Connection *> *>(&list)) {
84 if (c->type_ == Connection::Direct || (c->type_ != Connection::Queued && c->thread_->id() == std::this_thread::get_id())) {
85 (*c->f)(args...);
86 continue;
87 }
88 if (c->type_ != Connection::Sync) {
89 AbstractThread::AbstractTask *_t = new QueuedTask(c->f->copy(), args...);
90 if (!c->thread_->invokeTask(_t)) delete _t;
91 } else {
92 c->thread_->invokeMethod([c, &args...]() mutable { (*c->f)(args...); }, true);
93 }
94 }
95 }
