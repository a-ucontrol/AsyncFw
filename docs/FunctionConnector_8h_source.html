90 void operator()(Args... args) {
91 std::lock_guard<std::mutex> lock(mutex);
92 for (const Connection *c : *reinterpret_cast<std::vector<Connection *> *>(&list)) {
93 if (!c->thread_) continue;
94 if (c->type_ == Connection::Direct || (c->type_ != Connection::Queued && c->thread_->id() == std::this_thread::get_id())) {
95 (c->f_)->invoke(args...);
96 continue;
97 }
98 if (c->type_ != Connection::Sync) {
99 AbstractThread::AbstractTask *_t = new QueuedTask(c->f_->copy(), args...);
100 if (!c->thread_->invokeTask(_t)) delete _t;
101 } else {
102 c->thread_->invoke([c, &args...]() { (c->f_)->invoke(args...); }, true);
103 }
104 }
105 }
