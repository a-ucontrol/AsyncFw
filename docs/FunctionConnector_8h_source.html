114 void operator()(Args... args) const {
115 std::lock_guard<std::mutex> lock(mutex);
116 for (const AbstractFunctionConnector::Connection *c : list) {
117 if (!c->thread_) continue;
118 if (c->type_ == Connection::Direct || (c->type_ != Connection::Queued && c->thread_->id() == std::this_thread::get_id())) {
119 (static_cast<const Connection *>(c)->f_)->invoke(args...);
120 continue;
121 }
122 if (c->type_ != Connection::Sync) {
123 AbstractThread::AbstractTask *_t = new QueuedTask(static_cast<const Connection *>(c)->f_->copy(), args...);
124 if (!c->thread_->invokeTask(_t)) delete _t;
125 } else {
126 c->thread_->invoke([c, &args...]() { (static_cast<const Connection *>(c)->f_)->invoke(args...); }, true);
127 }

149 struct AbstractFunction : public AsyncFw::Invocable<void(Args &...)>::Abstract {
150 /*
151 Invokes the target callable, guaranteeing COPY semantics for arguments.
152 This method is explicitly used during immediate signal dispatching (Direct/Sync) within the loop.
153 It passes arguments as plain lvalues, forcing the compiler to generate independent copies for each distinct receiver slot.
154 The default operator() cannot be used for this purpose because it uses std::forward, which casts lvalues to rvalues (triggering move semantics).
155 If operator() were called in a loop, the very first receiver would empty out (steal resources from) the arguments, leaving subsequent subscribers with moved-from, empty objects.
156 */
157 virtual void invoke(Args &...args) = 0;
158 virtual AbstractFunction *copy() const = 0;
159 };
160 template <typename F>
161 struct Function : AbstractFunction {
162 Function(F &&f) : f_(std::forward<F>(f)) {}
163 Function(const Function *f) : f_(f->f_) {}
164 void operator()(Args &...args) override { f_(std::forward<Args>(args)...); }
165 void invoke(Args &...args) override { f_(args...); }
166 AbstractFunction *copy() const override { return new Function(this); }
167 F f_;
168 };
169 template <typename M, typename O>
170 struct MemberFunction : AbstractFunction {
171 MemberFunction(M m, O *o) : m_(m), o_(o) {}
172 MemberFunction(const MemberFunction *f) : m_(f->m_), o_(f->o_) {}
173 void operator()(Args &...args) override { (o_->*m_)(std::forward<Args>(args)...); }
174 void invoke(Args &...args) override { (o_->*m_)(args...); }
175 AbstractFunction *copy() const override { return new MemberFunction(this); }
176 M m_;
177 O *o_;
178 };
179 ~Connection() override { delete f_; }
180 AbstractFunction *f_;
181 };
182};
